use App\Modules\Page\Models\Page;
use App\Modules\Page\Models\PageSection;

public function page($slug = 'home')
    {
        if(config('page.hide-page.'.$slug)){
            abort(404);
        }
        $page = new Page();
        $page = $page->getPage($slug);

        $data = [];
        if ($slug == 'home') { //for home page values
            //masters
            $data['courseSubjects'] = CourseSubject::all();
            $data['courseTypes'] = CourseType::all();
            $data['degreeLevels'] = DegreeLevel::all();
        }

        $pagesections = PageSection::where('page_id', $page->id)->orderBy('order', 'ASC')->get()->groupBy('type');

        return view(config('page.blade.'.$page->blade), ['page' => $page, 'pagesections' => $pagesections, 'slug' => $slug, 'data' => $data]);
    }
